    message(FATAL_ERROR "glslc not found. Please install Vulkan SDK shader tools.")

set(WALK_REQUIRED_SHADER_SOURCES
    ${WALK_MODEL_VERTEX_SHADER}

    ${WALK_WALL_FRAGMENT_SHADER}

    ${WALK_POINT_PARTICLE_VERTEX_SHADER}
    ${WALK_POINT_PARTICLE_FRAGMENT_SHADER}
    ${WALK_BUBBLE_FRAGMENT_SHADER}
    ${WALK_FLOOR_KALE_FRAGMENT_SHADER}
    ${WALK_FLOOR_SWIRL_FRAGMENT_SHADER}
    ${WALK_FLOOR_TWIST_FRAGMENT_SHADER}
)

foreach(WALK_SHADER_FILE IN LISTS WALK_REQUIRED_SHADER_SOURCES)
    if(NOT EXISTS "${WALK_SHADER_FILE}")
        message(FATAL_ERROR
            "walk_post missing shader source: ${WALK_SHADER_FILE}. "
            "Ensure examples/walk_post/shaders is present and rerun CMake configure."
        )
    endif()
endforeach()

    COMMAND ${CMAKE_COMMAND} -E make_directory $<TARGET_FILE_DIR:walk_post>/data
    COMMAND ${CMAKE_COMMAND} -E copy_directory
        ${CMAKE_CURRENT_SOURCE_DIR}/data
        $<TARGET_FILE_DIR:walk_post>/data

        ${CMAKE_SOURCE_DIR}/models/cube.mxmod.z
        ${CMAKE_SOURCE_DIR}/models/sphere.mxmod.z

    COMMAND ${CMAKE_COMMAND} -E copy_if_different
