add_executable(starship
    ${CMAKE_SOURCE_DIR}/examples/starship/starship.cpp
)
target_link_libraries(starship PRIVATE mxvk::mxvk)

find_program(GLSLC_EXECUTABLE glslc)
if(NOT GLSLC_EXECUTABLE)
    message(FATAL_ERROR "glslc not found. Please install Vulkan SDK shader tools.")
endif()

set(STARSHIP_EXAMPLE_SHADER_SOURCE_DIR ${CMAKE_CURRENT_SOURCE_DIR}/shaders)
set(STARSHIP_EXAMPLE_SHADER_OUTPUT_DIR ${CMAKE_CURRENT_BINARY_DIR}/shaders)

set(STARSHIP_EXAMPLE_VERTEX_SHADER ${STARSHIP_EXAMPLE_SHADER_SOURCE_DIR}/model.vert)
set(STARSHIP_EXAMPLE_FRAGMENT_SHADER ${STARSHIP_EXAMPLE_SHADER_SOURCE_DIR}/model.frag)
set(STARSHIP_EXAMPLE_STAR_VERTEX_SHADER ${STARSHIP_EXAMPLE_SHADER_SOURCE_DIR}/star.vert)
set(STARSHIP_EXAMPLE_STAR_FRAGMENT_SHADER ${STARSHIP_EXAMPLE_SHADER_SOURCE_DIR}/star.frag)
set(STARSHIP_EXAMPLE_FLAME_VERTEX_SHADER ${STARSHIP_EXAMPLE_SHADER_SOURCE_DIR}/flame.vert)
set(STARSHIP_EXAMPLE_FLAME_FRAGMENT_SHADER ${STARSHIP_EXAMPLE_SHADER_SOURCE_DIR}/flame.frag)
set(STARSHIP_EXAMPLE_VERTEX_SPV ${STARSHIP_EXAMPLE_SHADER_OUTPUT_DIR}/model.vert.spv)
set(STARSHIP_EXAMPLE_FRAGMENT_SPV ${STARSHIP_EXAMPLE_SHADER_OUTPUT_DIR}/model.frag.spv)
set(STARSHIP_EXAMPLE_STAR_VERTEX_SPV ${STARSHIP_EXAMPLE_SHADER_OUTPUT_DIR}/star.vert.spv)
set(STARSHIP_EXAMPLE_STAR_FRAGMENT_SPV ${STARSHIP_EXAMPLE_SHADER_OUTPUT_DIR}/star.frag.spv)
set(STARSHIP_EXAMPLE_FLAME_VERTEX_SPV ${STARSHIP_EXAMPLE_SHADER_OUTPUT_DIR}/flame.vert.spv)
set(STARSHIP_EXAMPLE_FLAME_FRAGMENT_SPV ${STARSHIP_EXAMPLE_SHADER_OUTPUT_DIR}/flame.frag.spv)

add_custom_command(
    OUTPUT ${STARSHIP_EXAMPLE_VERTEX_SPV}
    COMMAND ${CMAKE_COMMAND} -E make_directory ${STARSHIP_EXAMPLE_SHADER_OUTPUT_DIR}
    COMMAND ${GLSLC_EXECUTABLE} ${STARSHIP_EXAMPLE_VERTEX_SHADER} -o ${STARSHIP_EXAMPLE_VERTEX_SPV}
    DEPENDS ${STARSHIP_EXAMPLE_VERTEX_SHADER}
    COMMENT "Compiling starship example vertex shader"
    VERBATIM
)

add_custom_command(
    OUTPUT ${STARSHIP_EXAMPLE_FRAGMENT_SPV}
    COMMAND ${CMAKE_COMMAND} -E make_directory ${STARSHIP_EXAMPLE_SHADER_OUTPUT_DIR}
    COMMAND ${GLSLC_EXECUTABLE} ${STARSHIP_EXAMPLE_FRAGMENT_SHADER} -o ${STARSHIP_EXAMPLE_FRAGMENT_SPV}
    DEPENDS ${STARSHIP_EXAMPLE_FRAGMENT_SHADER}
    COMMENT "Compiling starship example fragment shader"
    VERBATIM
)

add_custom_command(
    OUTPUT ${STARSHIP_EXAMPLE_STAR_VERTEX_SPV}
    COMMAND ${CMAKE_COMMAND} -E make_directory ${STARSHIP_EXAMPLE_SHADER_OUTPUT_DIR}
    COMMAND ${GLSLC_EXECUTABLE} ${STARSHIP_EXAMPLE_STAR_VERTEX_SHADER} -o ${STARSHIP_EXAMPLE_STAR_VERTEX_SPV}
    DEPENDS ${STARSHIP_EXAMPLE_STAR_VERTEX_SHADER}
    COMMENT "Compiling starship starfield vertex shader"
    VERBATIM
)

add_custom_command(
    OUTPUT ${STARSHIP_EXAMPLE_STAR_FRAGMENT_SPV}
    COMMAND ${CMAKE_COMMAND} -E make_directory ${STARSHIP_EXAMPLE_SHADER_OUTPUT_DIR}
    COMMAND ${GLSLC_EXECUTABLE} ${STARSHIP_EXAMPLE_STAR_FRAGMENT_SHADER} -o ${STARSHIP_EXAMPLE_STAR_FRAGMENT_SPV}
    DEPENDS ${STARSHIP_EXAMPLE_STAR_FRAGMENT_SHADER}
    COMMENT "Compiling starship starfield fragment shader"
    VERBATIM
)

add_custom_command(
    OUTPUT ${STARSHIP_EXAMPLE_FLAME_VERTEX_SPV}
    COMMAND ${CMAKE_COMMAND} -E make_directory ${STARSHIP_EXAMPLE_SHADER_OUTPUT_DIR}
    COMMAND ${GLSLC_EXECUTABLE} ${STARSHIP_EXAMPLE_FLAME_VERTEX_SHADER} -o ${STARSHIP_EXAMPLE_FLAME_VERTEX_SPV}
    DEPENDS ${STARSHIP_EXAMPLE_FLAME_VERTEX_SHADER}
    COMMENT "Compiling starship flame vertex shader"
    VERBATIM
)

add_custom_command(
    OUTPUT ${STARSHIP_EXAMPLE_FLAME_FRAGMENT_SPV}
    COMMAND ${CMAKE_COMMAND} -E make_directory ${STARSHIP_EXAMPLE_SHADER_OUTPUT_DIR}
    COMMAND ${GLSLC_EXECUTABLE} ${STARSHIP_EXAMPLE_FLAME_FRAGMENT_SHADER} -o ${STARSHIP_EXAMPLE_FLAME_FRAGMENT_SPV}
    DEPENDS ${STARSHIP_EXAMPLE_FLAME_FRAGMENT_SHADER}
    COMMENT "Compiling starship flame fragment shader"
    VERBATIM
)

add_custom_target(starship_shaders
    DEPENDS
        ${STARSHIP_EXAMPLE_VERTEX_SPV}
        ${STARSHIP_EXAMPLE_FRAGMENT_SPV}
        ${STARSHIP_EXAMPLE_STAR_VERTEX_SPV}
        ${STARSHIP_EXAMPLE_STAR_FRAGMENT_SPV}
        ${STARSHIP_EXAMPLE_FLAME_VERTEX_SPV}
        ${STARSHIP_EXAMPLE_FLAME_FRAGMENT_SPV}
)

add_dependencies(starship starship_shaders)
target_compile_definitions(starship PRIVATE
    STARSHIP_EXAMPLE_SHADER_DIR="${STARSHIP_EXAMPLE_SHADER_OUTPUT_DIR}"
    STARSHIP_EXAMPLE_ASSET_DIR="$<TARGET_FILE_DIR:starship>"
    STARSHIP_EXAMPLE_RUNTIME_DATA_DIR="$<TARGET_FILE_DIR:starship>/data"
)

set(STARSHIP_EXAMPLE_MODEL_SOURCE ${CMAKE_CURRENT_SOURCE_DIR}/data/starship.obj)
set(STARSHIP_EXAMPLE_MTL_SOURCE ${CMAKE_CURRENT_SOURCE_DIR}/data/starship.mtl)
set(STARSHIP_EXAMPLE_TEXTURE_SOURCES
    ${CMAKE_CURRENT_SOURCE_DIR}/data/rear_cap_metal.png
    ${CMAKE_CURRENT_SOURCE_DIR}/data/plane.png
    ${CMAKE_CURRENT_SOURCE_DIR}/data/ship_cockpit.png
    ${CMAKE_CURRENT_SOURCE_DIR}/data/ship_cockpit_red.png
    ${CMAKE_CURRENT_SOURCE_DIR}/data/ship_fin.png
    ${CMAKE_CURRENT_SOURCE_DIR}/data/ship_engine.png
    ${CMAKE_CURRENT_SOURCE_DIR}/data/ship_hull.png
    ${CMAKE_CURRENT_SOURCE_DIR}/data/ship_wing.png
)
set(STARSHIP_EXAMPLE_STAR_TEXTURE_SOURCE ${CMAKE_SOURCE_DIR}/examples/pong/data/star.png)
set(STARSHIP_EXAMPLE_RUNTIME_DATA_OUTPUT_DIR ${CMAKE_CURRENT_BINARY_DIR}/data)

foreach(_asset
    ${STARSHIP_EXAMPLE_MODEL_SOURCE}
    ${STARSHIP_EXAMPLE_MTL_SOURCE}
    ${STARSHIP_EXAMPLE_STAR_TEXTURE_SOURCE}
    ${STARSHIP_EXAMPLE_TEXTURE_SOURCES}
)
    if(NOT EXISTS ${_asset})
        message(FATAL_ERROR "starship example missing required asset: ${_asset}")
    endif()
endforeach()

add_custom_command(TARGET starship POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E make_directory ${STARSHIP_EXAMPLE_RUNTIME_DATA_OUTPUT_DIR}
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        ${STARSHIP_EXAMPLE_MODEL_SOURCE}
        ${STARSHIP_EXAMPLE_RUNTIME_DATA_OUTPUT_DIR}/starship.obj
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        ${STARSHIP_EXAMPLE_MTL_SOURCE}
        ${STARSHIP_EXAMPLE_RUNTIME_DATA_OUTPUT_DIR}/starship.mtl
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        ${STARSHIP_EXAMPLE_STAR_TEXTURE_SOURCE}
        ${STARSHIP_EXAMPLE_RUNTIME_DATA_OUTPUT_DIR}/star.png
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        ${STARSHIP_EXAMPLE_TEXTURE_SOURCES}
        ${STARSHIP_EXAMPLE_RUNTIME_DATA_OUTPUT_DIR}
    COMMENT "Copying starship example assets to runtime output directory"
)
