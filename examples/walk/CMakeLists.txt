add_executable(walk_example
    ${CMAKE_SOURCE_DIR}/examples/walk/room.cpp
)
target_link_libraries(walk_example PRIVATE mxvk::mxvk)

find_program(GLSLC_EXECUTABLE glslc)
if(NOT GLSLC_EXECUTABLE)
    message(FATAL_ERROR "glslc not found. Please install Vulkan SDK shader tools.")
endif()

set(WALK_SHADER_SOURCE_DIR ${CMAKE_CURRENT_SOURCE_DIR}/shaders)
set(WALK_SHADER_OUTPUT_DIR ${CMAKE_CURRENT_BINARY_DIR}/shaders)

set(WALK_MODEL_VERTEX_SHADER ${WALK_SHADER_SOURCE_DIR}/model.vert)
set(WALK_MODEL_FRAGMENT_SHADER ${WALK_SHADER_SOURCE_DIR}/model.frag)
set(WALK_PILLAR_VERTEX_SHADER ${WALK_SHADER_SOURCE_DIR}/pillar.vert)
set(WALK_WALL_FRAGMENT_SHADER ${WALK_SHADER_SOURCE_DIR}/wall.frag)
set(WALK_FLOOR_FRAGMENT_SHADER ${WALK_SHADER_SOURCE_DIR}/floor.frag)
set(WALK_PILLAR_FRAGMENT_SHADER ${WALK_SHADER_SOURCE_DIR}/pillar.frag)
set(WALK_OBJECT_FRAGMENT_SHADER ${WALK_SHADER_SOURCE_DIR}/object.frag)
set(WALK_BULLET_FRAGMENT_SHADER ${WALK_SHADER_SOURCE_DIR}/bullet.frag)
set(WALK_PARTICLE_FRAGMENT_SHADER ${WALK_SHADER_SOURCE_DIR}/particle.frag)
set(WALK_POINT_PARTICLE_VERTEX_SHADER ${WALK_SHADER_SOURCE_DIR}/particle_points.vert)
set(WALK_POINT_PARTICLE_FRAGMENT_SHADER ${WALK_SHADER_SOURCE_DIR}/particle_points.frag)
set(WALK_MODEL_VERTEX_SPV ${WALK_SHADER_OUTPUT_DIR}/model.vert.spv)
set(WALK_MODEL_FRAGMENT_SPV ${WALK_SHADER_OUTPUT_DIR}/model.frag.spv)
set(WALK_PILLAR_VERTEX_SPV ${WALK_SHADER_OUTPUT_DIR}/pillar.vert.spv)
set(WALK_WALL_FRAGMENT_SPV ${WALK_SHADER_OUTPUT_DIR}/wall.frag.spv)
set(WALK_FLOOR_FRAGMENT_SPV ${WALK_SHADER_OUTPUT_DIR}/floor.frag.spv)
set(WALK_PILLAR_FRAGMENT_SPV ${WALK_SHADER_OUTPUT_DIR}/pillar.frag.spv)
set(WALK_OBJECT_FRAGMENT_SPV ${WALK_SHADER_OUTPUT_DIR}/object.frag.spv)
set(WALK_BULLET_FRAGMENT_SPV ${WALK_SHADER_OUTPUT_DIR}/bullet.frag.spv)
set(WALK_PARTICLE_FRAGMENT_SPV ${WALK_SHADER_OUTPUT_DIR}/particle.frag.spv)
set(WALK_POINT_PARTICLE_VERTEX_SPV ${WALK_SHADER_OUTPUT_DIR}/particle_points.vert.spv)
set(WALK_POINT_PARTICLE_FRAGMENT_SPV ${WALK_SHADER_OUTPUT_DIR}/particle_points.frag.spv)

add_custom_command(
    OUTPUT ${WALK_MODEL_VERTEX_SPV}
    COMMAND ${CMAKE_COMMAND} -E make_directory ${WALK_SHADER_OUTPUT_DIR}
    COMMAND ${GLSLC_EXECUTABLE} ${WALK_MODEL_VERTEX_SHADER} -o ${WALK_MODEL_VERTEX_SPV}
    DEPENDS ${WALK_MODEL_VERTEX_SHADER}
    COMMENT "Compiling walk model vertex shader"
    VERBATIM
)

    OUTPUT ${WALK_MODEL_FRAGMENT_SPV}

add_custom_target(walk_example_shaders
    DEPENDS
        ${WALK_MODEL_VERTEX_SPV}
        ${WALK_MODEL_FRAGMENT_SPV}
        ${WALK_PILLAR_VERTEX_SPV}
    ${WALK_WALL_FRAGMENT_SPV}

    ${WALK_POINT_PARTICLE_FRAGMENT_SPV}
)

add_dependencies(walk_example walk_example_shaders)
target_compile_definitions(walk_example PRIVATE
    WALK_SHADER_DIR="${WALK_SHADER_OUTPUT_DIR}"
    WALK_ASSET_DIR="${CMAKE_CURRENT_SOURCE_DIR}"
    WALK_MODELS_DIR="${CMAKE_SOURCE_DIR}/models"
)

add_custom_command(TARGET walk_example POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E make_directory $<TARGET_FILE_DIR:walk_example>/data
    COMMAND ${CMAKE_COMMAND} -E copy_directory
        ${CMAKE_CURRENT_SOURCE_DIR}/data
        $<TARGET_FILE_DIR:walk_example>/data
    COMMENT "Copying walk assets to runtime output directory"
)
